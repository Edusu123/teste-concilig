CREATE TABLE Fabricante ( 
	id_Fabricante SERIAL NOT NULL, 
	nome varchar(50) NOT NULL,
	cnpj varchar(20) NOT NULL,
	PRIMARY KEY (id_Fabricante) 
); 

CREATE TABLE Aparelho ( 
	id_aparelho SERIAL NOT NULL, 
	id_fabricante int NOT NULL REFERENCES Fabricante (id_Fabricante), 
	modelo varchar(50) NOT NULL, 
	altura float NOT NULL, 
	largura float NOT NULL, 
	espessura float NOT NULL, 
	peso int NOT NULL, 
	preco float NOT NULL, 
	quantidade int NOT NULL, 
	desconto float, 
	PRIMARY KEY (id_aparelho)
);

CREATE TABLE Pedido(
	id_Pedido SERIAL NOT NULL,
	id_aparelho int NOT NULL REFERENCES Aparelho (id_aparelho),
	datahorapedido timestamp NOT NULL,
	observacao varchar(255),
	PRIMARY KEY (id_Pedido)
);
